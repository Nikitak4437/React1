Que3] You are given an integer N , and your task is to find the sum of all the even integers starting from 1 upto N ( N inclusive). 

        
const findSum = (n) => {
  
    var son=0;
    for(var i=0; i<=n; i++)
    {
      if (i%2==0)
      {
        son=son + i;
      }
    }
